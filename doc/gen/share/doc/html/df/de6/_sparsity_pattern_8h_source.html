71template <
72 common::CContiguousIndexRange TRowIndexRange,
73 common::CContiguousIndexRange TColIndexRange>
74inline void SparsityPattern::Compute(
75 Index nRows,
76 Index nCols,
77 TRowIndexRange&& rowIndices,
78 TColIndexRange&& colIndices)
79{
80 PBAT_PROFILE_NAMED_SCOPE("pbat.math.linalg.SparsityPattern.Compute");
81 namespace srng = std::ranges;
82 namespace sviews = std::views;
83 namespace rng = ranges;
84 namespace views = rng::views;
85
86 A = CSCMatrix(nRows, nCols);
87 ij.resize(srng::size(rowIndices), Index{0});
88
89 auto const [rowMin, rowMax] = srng::minmax_element(rowIndices);
90 auto const [colMin, colMax] = srng::minmax_element(colIndices);
91 bool const bRowsInBounds = (*rowMin >= 0) && (*rowMax < nRows);
92 bool const bColsInBounds = (*colMin >= 0) && (*colMax < nCols);
93 if (not(bRowsInBounds and bColsInBounds))
94 {
95 std::string const what = fmt::format(
96 "Out of bounds min (row,col)=({},{}) and max (row,col)=({},{}), while "
97 "(nRows,nCols)=({},{})",
98 *rowMin,
99 *colMin,
100 *rowMax,
101 *rowMax,
102 nRows,
103 nCols);
104 throw std::invalid_argument(what);
105 }
106
107 auto const nRowIndices = srng::size(rowIndices);
108 auto const nColIndices = srng::size(colIndices);
109 if (nRowIndices != nColIndices)
110 {
111 std::string const what = fmt::format(
112 "Expected same number of row indices and column indices, but got {} row indices "
113 "and {} column indices",
114 nRowIndices,
115 nColIndices);
116 throw std::invalid_argument(what);
117 }
118 auto rows = srng::data(rowIndices);
119 auto cols = srng::data(colIndices);
120
121 auto const indexPair = [&](Index s) {
122 return std::make_pair(cols[s], rows[s]);
123 };
124 auto const less = [&](Index lhs, Index rhs) {
125 return indexPair(lhs) < indexPair(rhs);
126 };
127 auto const equal = [&](Index lhs, Index rhs) {
128 return indexPair(lhs) == indexPair(rhs);
129 };
130 auto const numNonZeroIndices = static_cast<Index>(nRowIndices);
131 // NOTE: Bottleneck is the sort
132 auto const sortedNonZeroIndices = views::iota(Index{0}, numNonZeroIndices) |
133 rng::to<std::vector>() | rng::actions::sort(less);
134 auto const sortedUniqueNonZeroIndices =
135 sortedNonZeroIndices | views::unique(equal) | rng::to<std::vector>();
136
137 IndexVectorX cc = IndexVectorX::Zero(nCols);
138 for (auto u : sortedUniqueNonZeroIndices)
139 ++cc[cols[u]];
140
141 for (auto k = 0, u = 0; k < numNonZeroIndices; ++k)
142 {
143 // NOTE: Yes, the casting is just absurd here... otherwise code
144 // doesn't compile with agressive warnings on signed/unsigned mismatch
145 auto const s = sortedNonZeroIndices[static_cast<std::size_t>(k)];
146 auto const uu = sortedUniqueNonZeroIndices[static_cast<std::size_t>(u)];
147 bool const bIsSameEntry = equal(s, uu);
148 ij[static_cast<std::size_t>(s)] = static_cast<Index>(bIsSameEntry ? u : ++u);
149 }
150 A.reserve(cc);
151 for (auto s : sortedUniqueNonZeroIndices)
152 A.insert(rows[s], cols[s]) = 0.;
153 A.makeCompressed();
154}

156template <common::CArithmeticRange TNonZeroRange>
157CSCMatrix SparsityPattern::ToMatrix(TNonZeroRange&& nonZeros) const
158{
159 PBAT_PROFILE_NAMED_SCOPE("pbat.math.linalg.SparsityPattern.ToMatrix");
160 static_assert(
161 std::is_same_v<Scalar, std::ranges::range_value_t<TNonZeroRange>>,
162 "Only Scalar non-zero values are accepted");
163
164 namespace rng = std::ranges;
165 auto const nnz = rng::size(nonZeros);
166 if (nnz != ij.size())
167 {
168 std::string const what = fmt::format("Expected {} non zeros, got {}", ij.size(), nnz);
169 throw std::invalid_argument(what);
170 }
171
172 CSCMatrix Acpy{A};
173 Scalar* values = Acpy.valuePtr();
174 for (auto k = 0ULL; k < nnz; ++k)
175 values[ij[k]] += nonZeros[k];
176 return Acpy;
177}
