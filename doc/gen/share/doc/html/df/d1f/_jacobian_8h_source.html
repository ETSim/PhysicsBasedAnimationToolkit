40template <CElement TElement, class TDerived>
41[[maybe_unused]] auto
42Jacobian(Vector<TElement::kDims> const& X, Eigen::MatrixBase<TDerived> const& x)
43 -> Matrix<TDerived::RowsAtCompileTime, TElement::kDims>
44{
45 assert(x.cols() == TElement::kNodes);
46 auto constexpr kDimsOut = TDerived::RowsAtCompileTime;
47 Matrix<kDimsOut, TElement::kDims> const J = x * TElement::GradN(X);

61template <class TDerived>
62[[maybe_unused]] Scalar DeterminantOfJacobian(Eigen::MatrixBase<TDerived> const& J)
63{
64 bool const bIsSquare = J.rows() == J.cols();
65 Scalar const detJ = bIsSquare ? J.determinant() : J.jacobiSvd().singularValues().prod();
66 auto constexpr eps = 1e-10;
67 if (detJ <= eps)
68 {
69 throw std::invalid_argument("Inverted or singular jacobian");
70 }
71 return detJ;
72}

83template <int QuadratureOrder, CMesh TMesh>
84MatrixX DeterminantOfJacobian(TMesh const& mesh)
85{
86 PBAT_PROFILE_NAMED_SCOPE("pbat.fem.DeterminantOfJacobian");
87
88 using ElementType = typename TMesh::ElementType;
89 using AffineElementType = typename ElementType::AffineBaseType;
90 using QuadratureRuleType = typename ElementType::template QuadratureType<QuadratureOrder>;
91
92 auto const Xg = common::ToEigen(QuadratureRuleType::points)
93 .reshaped(QuadratureRuleType::kDims + 1, QuadratureRuleType::kPoints)
94 .template bottomRows<ElementType::kDims>();
95
96 auto const numberOfElements = mesh.E.cols();
97 MatrixX detJe(QuadratureRuleType::kPoints, numberOfElements);
98 tbb::parallel_for(Index{0}, Index{numberOfElements}, [&](Index e) {
99 auto const nodes = mesh.E.col(e);
100 auto const vertices = nodes(ElementType::Vertices);
101 auto constexpr kRowsJ = TMesh::kDims;
102 auto constexpr kColsJ = AffineElementType::kNodes;
103 Matrix<kRowsJ, kColsJ> const Ve = mesh.X(Eigen::placeholders::all, vertices);
104 if constexpr (AffineElementType::bHasConstantJacobian)
105 {
106 Scalar const detJ = DeterminantOfJacobian(Jacobian<AffineElementType>({}, Ve));
107 detJe.col(e).setConstant(detJ);
108 }
109 else
110 {
111 for (auto g = 0; g < QuadratureRuleType::kPoints; ++g)
112 {
113 Scalar const detJ =
114 DeterminantOfJacobian(Jacobian<AffineElementType>(Xg.col(g), Ve));
115 detJe(g, e) = detJ;
116 }
117 }
118 });
119 return detJe;
120}

137MatrixX InnerProductWeights(TMesh const& mesh)

164 auto const wg = common::ToEigen(QuadratureRuleType::weights);

207template <CElement TElement, class TDerivedX, class TDerivedx>
208Vector<TElement::kDims> ReferencePosition(
209 Eigen::MatrixBase<TDerivedX> const& X,
210 Eigen::MatrixBase<TDerivedx> const& x,
211 int const maxIterations = 5,
212 Scalar const eps = 1e-10)
213{
214 using ElementType = TElement;
215 assert(x.cols() == ElementType::kNodes);
216 auto constexpr kDims = ElementType::kDims;
217 auto constexpr kNodes = ElementType::kNodes;
218 auto constexpr kOutDims = Eigen::MatrixBase<TDerivedx>::RowsAtCompileTime;
219 // Initial guess is element's barycenter.
220 auto const coords = common::ToEigen(ElementType::Coordinates).reshaped(kDims, kNodes);
221 auto const vertexLagrangePositions =
222 (coords(Eigen::placeholders::all, ElementType::Vertices).template cast<Scalar>() /
223 static_cast<Scalar>(ElementType::kOrder));
224 Vector<kDims> Xik =
225 vertexLagrangePositions.rowwise().sum() / static_cast<Scalar>(ElementType::Vertices.size());
226
227 Vector<kOutDims> rk = x * ElementType::N(Xik) - X;
228 Matrix<kOutDims, kDims> J;
229 Eigen::LDLT<Matrix<kDims, kDims>> LDLT;
230
231 // If Jacobian is constant, let's precompute it
232 if constexpr (ElementType::bHasConstantJacobian)
233 {
234 J = Jacobian<ElementType>(Xik, x);
235 LDLT.compute(J.transpose() * J);
236 }
237 // Do up to maxIterations Gauss Newton iterations
238 for (auto k = 0; k < maxIterations; ++k)
239 {
240 if (rk.template lpNorm<1>() <= eps)
241 break;
242
243 // Non-constant jacobians need to be updated
244 if constexpr (not ElementType::bHasConstantJacobian)
245 {
246 J = Jacobian<ElementType>(Xik, x);
247 LDLT.compute(J.transpose() * J);
248 }
249
250 Xik -= LDLT.solve(J.transpose() * rk);
251 rk = x * ElementType::N(Xik) - X;
252 }
253 return Xik;
254}

272template <CMesh TMesh, class TDerivedE, class TDerivedX>
273MatrixX ReferencePositions(
274 TMesh const& mesh,
275 Eigen::DenseBase<TDerivedE> const& E,
276 Eigen::MatrixBase<TDerivedX> const& X,
277 int maxIterations = 5,
278 Scalar eps = 1e-10)
279{
280 PBAT_PROFILE_NAMED_SCOPE("pbat.fem.ReferencePositions");
281 using MeshType = TMesh;
282 using ElementType = typename MeshType::ElementType;
283 MatrixX Xi(ElementType::kDims, E.size());
284 tbb::parallel_for(Index{0}, Index{E.size()}, [&](Index k) {
285 Index const e = E(k);
286 auto const nodes = mesh.E.col(e);
287 auto constexpr kOutDims = MeshType::kDims;
288 auto constexpr kNodes = ElementType::kNodes;
289 Matrix<kOutDims, kNodes> const Xe = mesh.X(Eigen::placeholders::all, nodes);
290 Vector<kOutDims> const Xk = X.col(k);
291 Xi.col(k) = ReferencePosition<ElementType>(Xk, Xe, maxIterations, eps);
292 });
293 return Xi;
294}
