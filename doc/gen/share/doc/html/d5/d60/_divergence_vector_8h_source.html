64template <CMesh TMesh, int Dims, int QuadratureOrder>
65template <class TDerived>
66inline DivergenceVector<TMesh, Dims, QuadratureOrder>::DivergenceVector(
67 MeshType const& meshIn,

93template <CMesh TMesh, int Dims, int QuadratureOrder>
94template <class TDerivedF>
95inline void DivergenceVector<TMesh, Dims, QuadratureOrder>::ComputeElementDivergence(
96 Eigen::DenseBase<TDerivedF> const& Fe)
97{
98 PBAT_PROFILE_NAMED_SCOPE("pbat.fem.DivergenceVector.ComputeElementDivergence");
99 CheckValidState();
100 auto const numberOfElements = mesh.E.cols();
101 divE.setZero(ElementType::kNodes, numberOfElements);
102 auto const wg = common::ToEigen(QuadratureRuleType::weights);
103 tbb::parallel_for(Index{0}, Index{numberOfElements}, [&](Index e) {
104 auto const nodes = mesh.E.col(e);
105 for (auto g = 0; g < QuadratureRuleType::kPoints; ++g)
106 {
107 auto constexpr kStride = MeshType::kDims * QuadratureRuleType::kPoints;
108 auto constexpr kNodesPerElement = ElementType::kNodes;
109 auto const gradPhi =
110 GNe.block<kNodesPerElement, MeshType::kDims>(0, e * kStride + g * MeshType::kDims);
111 auto const F = Fe(Eigen::placeholders::all, nodes);
112 // div(F) = \sum_i \sum_d F_id d(\phi_i) / d(X_d)
113 divE.col(e) =
114 (wg(g) * detJe(g, e)) * (F.array().transpose() * gradPhi.array()).rowwise().sum();
115 }
116 });
117}

119template <CMesh TMesh, int Dims, int QuadratureOrder>
120inline VectorX DivergenceVector<TMesh, Dims, QuadratureOrder>::ToVector() const
121{
122 PBAT_PROFILE_NAMED_SCOPE("pbat.fem.DivergenceVector.ToVector");
123 auto const numberOfNodes = mesh.X.cols();
124 auto const numberOfElements = mesh.E.cols();
125 VectorX div = VectorX::Zero(numberOfNodes);
126 for (auto e = 0; e < numberOfElements; ++e)
127 {
128 auto const nodes = mesh.E.col(e);
129 div(nodes) += divE.col(e);
130 }
131 return div;
132}

134template <CMesh TMesh, int Dims, int QuadratureOrder>
135inline void DivergenceVector<TMesh, Dims, QuadratureOrder>::CheckValidState() const
136{
137 auto const numberOfElements = mesh.E.cols();
138 auto constexpr kExpectedDetJeRows = QuadratureRuleType::kPoints;
139 auto const expectedDetJeCols = numberOfElements;
140 bool const bDeterminantsHaveCorrectDimensions =
141 (detJe.rows() == kExpectedDetJeRows) and (detJe.cols() == expectedDetJeCols);
142 if (not bDeterminantsHaveCorrectDimensions)
143 {
144 std::string const what = fmt::format(
145 "Expected determinants at element quadrature points of dimensions #quad.pts.={} x "
146 "#elements={} for polynomial "
147 "quadrature order={}, but got {}x{} instead.",
148 kExpectedDetJeRows,
149 expectedDetJeCols,
150 QuadratureOrder,
151 detJe.rows(),
152 detJe.cols());
153 throw std::invalid_argument(what);
154 }
155 auto constexpr kExpectedGNeRows = ElementType::kNodes;
156 auto const expectedGNeCols = MeshType::kDims * QuadratureRuleType::kPoints * numberOfElements;
157 bool const bShapeFunctionGradientsHaveCorrectDimensions =
158 (GNe.rows() == kExpectedGNeRows) and (GNe.cols() == expectedGNeCols);
159 if (not bShapeFunctionGradientsHaveCorrectDimensions)
160 {
161 std::string const what = fmt::format(
162 "Expected shape function gradients at element quadrature points of dimensions "
163 "|#nodes-per-element|={} x |#mesh-dims * #quad.pts. * #elemens|={} for polynomiail "
164 "quadrature order={}, but got {}x{} instead",
165 kExpectedGNeRows,
166 expectedGNeCols,
167 QuadratureOrder,
168 GNe.rows(),
169 GNe.cols());
170 throw std::invalid_argument(what);
171 }
172}
