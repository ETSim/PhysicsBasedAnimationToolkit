170template <CMesh TMesh, int QuadratureOrder>
171template <class TDerivedIn, class TDerivedOut>
172inline void MassMatrix<TMesh, QuadratureOrder>::Apply(
173 Eigen::MatrixBase<TDerivedIn> const& x,
174 Eigen::DenseBase<TDerivedOut>& y) const
175{
176 PBAT_PROFILE_NAMED_SCOPE("pbat.fem.MassMatrix.Apply");
177 CheckValidState();
178 auto const numberOfDofs = InputDimensions();
179 if (x.rows() != numberOfDofs or y.rows() != numberOfDofs or x.cols() != y.cols())
180 {
181 std::string const what = fmt::format(
182 "Expected inputs and outputs to have rows |#nodes*dims|={} and same number of "
183 "columns, but got dimensions "
184 "x,y=({},{}), ({},{})",
185 numberOfDofs,
186 x.rows(),
187 x.cols(),
188 y.rows(),
189 y.cols());
190 throw std::invalid_argument(what);
191 }
192
193 auto const numberOfElements = mesh.E.cols();
194 // NOTE: Could parallelize over columns, if there are many.
195 for (auto c = 0; c < y.cols(); ++c)
196 {
197 for (auto e = 0; e < numberOfElements; ++e)
198 {
199 auto const nodes = mesh.E.col(e).array();
200 auto const me =
201 Me.block<ElementType::kNodes, ElementType::kNodes>(0, e * ElementType::kNodes);
202 auto ye = y.col(c).reshaped(dims, y.size() / dims)(Eigen::placeholders::all, nodes);
203 auto const xe =
204 x.col(c).reshaped(dims, x.size() / dims)(Eigen::placeholders::all, nodes);
205 ye += xe * me /*.transpose() technically, but mass matrix is symmetric*/;
206 }
207 }
208}

210template <CMesh TMesh, int QuadratureOrder>
211inline CSCMatrix MassMatrix<TMesh, QuadratureOrder>::ToMatrix() const
212{
213 PBAT_PROFILE_NAMED_SCOPE("pbat.fem.MassMatrix.ToMatrix");
214 CheckValidState();
215 using SparseIndex = typename CSCMatrix::StorageIndex;
216 using Triplet = Eigen::Triplet<Scalar, SparseIndex>;
217
218 std::vector<Triplet> triplets{};
219 triplets.reserve(static_cast<std::size_t>(Me.size() * dims));
220 auto const numberOfElements = mesh.E.cols();
221 for (auto e = 0; e < numberOfElements; ++e)
222 {
223 auto const nodes = mesh.E.col(e);
224 auto const me =
225 Me.block<ElementType::kNodes, ElementType::kNodes>(0, e * ElementType::kNodes);
226 for (auto j = 0; j < me.cols(); ++j)
227 {
228 for (auto i = 0; i < me.rows(); ++i)
229 {
230 for (auto d = 0; d < dims; ++d)
231 {
232 auto const ni = static_cast<SparseIndex>(dims * nodes(i) + d);
233 auto const nj = static_cast<SparseIndex>(dims * nodes(j) + d);
234 triplets.push_back(Triplet(ni, nj, me(i, j)));
235 }
236 }
237 }
238 }
239
240 CSCMatrix Mmat(OutputDimensions(), InputDimensions());
241 Mmat.setFromTriplets(triplets.begin(), triplets.end());
242 return Mmat;
243}

245template <CMesh TMesh, int QuadratureOrder>
246inline VectorX MassMatrix<TMesh, QuadratureOrder>::ToLumpedMasses() const
247{
248 VectorX m(InputDimensions());
249 m.setZero();
250 auto const numberOfElements = mesh.E.cols();
251 for (auto e = 0; e < numberOfElements; ++e)
252 {
253 auto const nodes = mesh.E.col(e);
254 auto const me =
255 Me.block<ElementType::kNodes, ElementType::kNodes>(0, e * ElementType::kNodes);
256 for (auto j = 0; j < me.cols(); ++j)
257 {
258 for (auto i = 0; i < me.rows(); ++i)
259 {
260 for (auto d = 0; d < dims; ++d)
261 {
262 auto const ni = dims * nodes(i) + d;
263 m(ni) += me(i, j);
264 }
265 }
266 }
267 }
268 return m;
269}

271template <CMesh TMesh, int QuadratureOrder>
272inline void MassMatrix<TMesh, QuadratureOrder>::CheckValidState() const
273{
274 auto const numberOfElements = mesh.E.cols();
275 auto constexpr kExpectedDetJeRows = QuadratureRuleType::kPoints;
276 auto const expectedDetJeCols = numberOfElements;
277 bool const bDeterminantsHaveCorrectDimensions =
278 (detJe.rows() == kExpectedDetJeRows) and (detJe.cols() == expectedDetJeCols);
279 if (not bDeterminantsHaveCorrectDimensions)
280 {
281 std::string const what = fmt::format(
282 "Expected determinants at element quadrature points of dimensions #quad.pts.={} x "
283 "#elements={} for polynomial "
284 "quadrature order={}, but got {}x{} instead.",
285 kExpectedDetJeRows,
286 expectedDetJeCols,
287 QuadratureOrder,
288 detJe.rows(),
289 detJe.cols());
290 throw std::invalid_argument(what);
291 }
292 if (dims < 1)
293 {
294 std::string const what =
295 fmt::format("Expected output dimensionality >= 1, got {} instead", dims);
296 throw std::invalid_argument(what);
297 }
298}

300template <CMesh TMesh, int QuadratureOrder>
301template <class TDerived>
302inline void MassMatrix<TMesh, QuadratureOrder>::ComputeElementMassMatrices(
303 Eigen::DenseBase<TDerived> const& rho)
304{
305 PBAT_PROFILE_NAMED_SCOPE("pbat.fem.MassMatrix.ComputeElementMassMatrices");
306 CheckValidState();
307 auto const numberOfElements = mesh.E.cols();
308 auto constexpr kNodesPerElement = ElementType::kNodes;
309 auto constexpr kQuadPtsPerElement = QuadratureRuleType::kPoints;
310 bool const bRhoDimensionsAreCorrect =
311 (rho.cols() == numberOfElements) and (rho.rows() == kQuadPtsPerElement);
312 if (not bRhoDimensionsAreCorrect)
313 {
314 std::string const what = fmt::format(
315 "Expected mass density rho of dimensions {}x{}, but dimensions were "
316 "{}x{}",
317 kQuadPtsPerElement,
318 numberOfElements,
319 rho.rows(),
320 rho.cols());
321 throw std::invalid_argument(what);
322 }
323 // Precompute element shape function outer products
324 auto const N = ShapeFunctions<ElementType, kQuadratureOrder>();
325 std::array<Matrix<kNodesPerElement, kNodesPerElement>, kQuadPtsPerElement> NgOuterNg{};
326 auto const wg = common::ToEigen(QuadratureRuleType::weights);
327 for (auto g = 0; g < kQuadPtsPerElement; ++g)
328 {
329 NgOuterNg[static_cast<std::size_t>(g)] = wg(g) * (N.col(g) * N.col(g).transpose());
330 }
331 // Compute element mass matrices
332 Me.setZero(kNodesPerElement, kNodesPerElement * numberOfElements);
333 tbb::parallel_for(Index{0}, Index{numberOfElements}, [&](Index e) {
334 auto me = Me.block<kNodesPerElement, kNodesPerElement>(0, e * kNodesPerElement);
335 for (auto g = 0; g < kQuadPtsPerElement; ++g)
336 {
337 me += (rho(g, e) * detJe(g, e)) * NgOuterNg[static_cast<std::size_t>(g)];
338 }
339 });
340}
