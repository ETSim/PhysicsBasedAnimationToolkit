329template <CMesh TMesh, physics::CHyperElasticEnergy THyperElasticEnergy>
330template <class TDerived>
331inline void HyperElasticPotential<TMesh, THyperElasticEnergy>::ComputeElementElasticity(
332 Eigen::MatrixBase<TDerived> const& x,
333 bool bWithGradient,
334 bool bWithHessian,
335 bool bUseSpdProjection)
336{
337 PBAT_PROFILE_NAMED_SCOPE("pbat.fem.HyperElasticPotential.ComputeElementElasticity");
338 // Check inputs
339 CheckValidState();
340 auto const numberOfNodes = mesh.X.cols();
341 if (x.size() != numberOfNodes * kDims)
342 {
343 std::string const what = fmt::format(
344 "Generalized coordinate vector must have dimensions |#nodes|*kDims={}, but got "
345 "x.size()={}",
346 numberOfNodes * kDims,
347 x.size());
348 throw std::invalid_argument(what);
349 }
350
351 Ug.setZero();
352 if (bWithGradient)
353 Gg.setZero();
354 if (bWithHessian)
355 Hg.setZero();
356
357 ElasticEnergyType Psi{};
358
359 // Compute element elastic energies and their derivatives
360 auto const numberOfQuadraturePoints = wg.size();
361 auto constexpr kNodesPerElement = ElementType::kNodes;
362 auto constexpr kDofsPerElement = kNodesPerElement * kDims;
363 namespace mini = math::linalg::mini;
364 using mini::FromEigen;
365 using mini::ToEigen;
366 if (not bWithGradient and not bWithHessian)
367 {
368 tbb::parallel_for(Index{0}, Index{numberOfQuadraturePoints}, [&](Index g) {
369 auto const e = eg(g);
370 auto const nodes = mesh.E.col(e);
371 auto const xe = x.reshaped(kDims, numberOfNodes)(Eigen::placeholders::all, nodes);
372 auto const GPeg = GNeg.block<kNodesPerElement, MeshType::kDims>(0, g * MeshType::kDims);
373 Matrix<kDims, kDims> const F = xe * GPeg;
374 auto vecF = FromEigen(F);
375 auto psiF = Psi.eval(vecF, mug(g), lambdag(g));
376 Ug(g) += wg(g) * psiF;
377 });
378 }
379 else if (bWithGradient and not bWithHessian)
380 {
381 tbb::parallel_for(Index{0}, Index{numberOfQuadraturePoints}, [&](Index g) {
382 auto const e = eg(g);
383 auto const nodes = mesh.E.col(e);
384 auto const xe = x.reshaped(kDims, numberOfNodes)(Eigen::placeholders::all, nodes);
385 auto const GPeg = GNeg.block<kNodesPerElement, MeshType::kDims>(0, g * MeshType::kDims);
386 Matrix<kDims, kDims> const F = xe * GPeg;
387 auto vecF = FromEigen(F);
388 mini::SVector<Scalar, kDims * kDims> gradPsiF;
389 auto psiF = Psi.evalWithGrad(vecF, mug(g), lambdag(g), gradPsiF);
390 Ug(g) += wg(g) * psiF;
391 auto const GP = FromEigen(GPeg);
392 auto GPsix = GradientWrtDofs<ElementType, kDims>(gradPsiF, GP);
393 Gg.col(g) += wg(g) * ToEigen(GPsix);
394 });
395 }
396 else if (not bWithGradient and bWithHessian)
397 {
398 tbb::parallel_for(Index{0}, Index{numberOfQuadraturePoints}, [&](Index g) {
399 auto const e = eg(g);
400 auto const nodes = mesh.E.col(e);
401 auto const xe = x.reshaped(kDims, numberOfNodes)(Eigen::placeholders::all, nodes);
402 auto const gradPhi =
403 GNeg.block<kNodesPerElement, MeshType::kDims>(0, g * MeshType::kDims);
404 Matrix<kDims, kDims> const F = xe * gradPhi;
405 auto vecF = FromEigen(F);
406 auto psiF = Psi.eval(vecF, mug(g), lambdag(g));
407 auto hessPsiF = Psi.hessian(vecF, mug(g), lambdag(g));
408 Ug(g) += wg(g) * psiF;
409 auto const GP = FromEigen(gradPhi);
410 auto HPsix = HessianWrtDofs<ElementType, kDims>(hessPsiF, GP);
411 auto heg = Hg.block<kDofsPerElement, kDofsPerElement>(0, g * kDofsPerElement);
412 heg += wg(g) * ToEigen(HPsix);
413 });
414 }
415 else
416 {
417 tbb::parallel_for(Index{0}, Index{numberOfQuadraturePoints}, [&](Index g) {
418 auto const e = eg(g);
419 auto const nodes = mesh.E.col(e);
420 auto const xe = x.reshaped(kDims, numberOfNodes)(Eigen::placeholders::all, nodes);
421 auto const GPeg = GNeg.block<kNodesPerElement, MeshType::kDims>(0, g * MeshType::kDims);
422 Matrix<kDims, kDims> const F = xe * GPeg;
423 auto vecF = FromEigen(F);
424 mini::SVector<Scalar, kDims * kDims> gradPsiF;
425 mini::SMatrix<Scalar, kDims * kDims, kDims * kDims> hessPsiF;
426 auto psiF = Psi.evalWithGradAndHessian(vecF, mug(g), lambdag(g), gradPsiF, hessPsiF);
427 auto const GP = FromEigen(GPeg);
428 auto GPsix = GradientWrtDofs<ElementType, kDims>(gradPsiF, GP);
429 auto HPsix = HessianWrtDofs<ElementType, kDims>(hessPsiF, GP);
430 auto heg = Hg.block<kDofsPerElement, kDofsPerElement>(0, g * kDofsPerElement);
431 Ug(g) += wg(g) * psiF;
432 Gg.col(g) += wg(g) * ToEigen(GPsix);
433 heg += wg(g) * ToEigen(HPsix);
434 });
435 }
436 if (bWithHessian and bUseSpdProjection)
437 {
438 tbb::parallel_for(Index{0}, Index{numberOfQuadraturePoints}, [&](Index g) {
439 auto heg = Hg.block<kDofsPerElement, kDofsPerElement>(0, g * kDofsPerElement);
440 Eigen::JacobiSVD<
441 Matrix<kDofsPerElement, kDofsPerElement>,
442 Eigen::ComputeFullU | Eigen::ComputeFullV>
443 SVD{};
444 SVD.compute(heg);
445 Vector<kDofsPerElement> sigma = SVD.singularValues();
446 for (auto s = sigma.size() - 1; s >= 0; --s)
447 {
448 if (sigma(s) >= 0.)
449 break;
450 sigma(s) = -sigma(s);
451 }
452 heg = SVD.matrixU() * sigma.asDiagonal() * SVD.matrixV().transpose();
453 });
454 }
455}

457template <CMesh TMesh, physics::CHyperElasticEnergy THyperElasticEnergy>
458template <class TDerivedIn, class TDerivedOut>
459inline void HyperElasticPotential<TMesh, THyperElasticEnergy>::Apply(
460 Eigen::MatrixBase<TDerivedIn> const& x,
461 Eigen::DenseBase<TDerivedOut>& y) const
462{
463 PBAT_PROFILE_NAMED_SCOPE("pbat.fem.HyperElasticPotential.Apply");
464 auto const numberOfDofs = InputDimensions();
465 if (x.rows() != numberOfDofs or y.rows() != numberOfDofs or x.cols() != y.cols())
466 {
467 std::string const what = fmt::format(
468 "Expected inputs and outputs to have rows |#nodes*kDims|={} and same number of "
469 "columns, but got dimensions "
470 "x,y=({},{}), ({},{})",
471 numberOfDofs,
472 x.rows(),
473 x.cols(),
474 y.rows(),
475 y.cols());
476 throw std::invalid_argument(what);
477 }
478
479 auto constexpr kDofsPerElement = kDims * ElementType::kNodes;
480 auto const numberOfQuadraturePoints = wg.size();
481 // NOTE: Outer loop could be parallelized over columns, and using graph coloring, inner loop
482 // could also be parallelized, if it's worth it.
483 for (auto c = 0; c < x.cols(); ++c)
484 {
485 for (auto g = 0; g < numberOfQuadraturePoints; ++g)
486 {
487 auto const e = eg(g);
488 auto const nodes = mesh.E.col(e);
489 auto const heg = Hg.block<kDofsPerElement, kDofsPerElement>(0, g * kDofsPerElement);
490 auto const xe =
491 x.col(c).reshaped(kDims, x.size() / kDims)(Eigen::placeholders::all, nodes);
492 auto ye = y.col(c).reshaped(kDims, y.size() / kDims)(Eigen::placeholders::all, nodes);
493 ye.reshaped() += heg * xe.reshaped();
494 }
495 }
496}

498template <CMesh TMesh, physics::CHyperElasticEnergy THyperElasticEnergy>
499inline void HyperElasticPotential<TMesh, THyperElasticEnergy>::PrecomputeHessianSparsity()
500{
501 PBAT_PROFILE_NAMED_SCOPE("pbat.fem.HyperElasticPotential.PrecomputeHessianSparsity");
502 auto const numberOfQuadraturePoints = wg.size();
503 auto const kNodesPerElement = ElementType::kNodes;
504 auto const kDofsPerElement = kNodesPerElement * kDims;
505 std::vector<Index> nonZeroRowIndices{};
506 std::vector<Index> nonZeroColIndices{};
507 nonZeroRowIndices.reserve(
508 static_cast<std::size_t>(kDofsPerElement * kDofsPerElement * numberOfQuadraturePoints));
509 nonZeroColIndices.reserve(
510 static_cast<std::size_t>(kDofsPerElement * kDofsPerElement * numberOfQuadraturePoints));
511 // Insert non-zero indices in the storage order of our Hg matrix of element hessians at
512 // quadrature points
513 for (auto g = 0; g < numberOfQuadraturePoints; ++g)
514 {
515 auto const e = eg(g);
516 auto const nodes = mesh.E.col(e);
517 for (auto j = 0; j < kNodesPerElement; ++j)
518 {
519 for (auto dj = 0; dj < kDims; ++dj)
520 {
521 for (auto i = 0; i < kNodesPerElement; ++i)
522 {
523 for (auto di = 0; di < kDims; ++di)
524 {
525 nonZeroRowIndices.push_back(kDims * nodes(i) + di);
526 nonZeroColIndices.push_back(kDims * nodes(j) + dj);
527 }
528 }
529 }
530 }
531 }
532 GH.Compute(OutputDimensions(), InputDimensions(), nonZeroRowIndices, nonZeroColIndices);
533}

535template <CMesh TMesh, physics::CHyperElasticEnergy THyperElasticEnergy>
536inline CSCMatrix HyperElasticPotential<TMesh, THyperElasticEnergy>::ToMatrix() const
537{
538 PBAT_PROFILE_NAMED_SCOPE("pbat.fem.HyperElasticPotential.ToMatrix");
539 if (!GH.IsEmpty())
540 {
541 using SpanType = std::span<Scalar const>;
542 using SizeType = typename SpanType::size_type;
543 return GH.ToMatrix(SpanType(Hg.data(), static_cast<SizeType>(Hg.size())));
544 }
545 else
546 {
547 // Construct hessian from triplets
548 using SparseIndex = typename CSCMatrix::StorageIndex;
549 using Triplet = Eigen::Triplet<Scalar, SparseIndex>;
550 std::vector<Triplet> triplets{};
551 triplets.reserve(static_cast<std::size_t>(Hg.size()));
552 auto const numberOfQuadraturePoints = wg.size();
553 for (auto g = 0; g < numberOfQuadraturePoints; ++g)
554 {
555 auto const e = eg(g);
556 auto const nodes = mesh.E.col(e);
557 auto constexpr Hrows = ElementType::kNodes * kDims;
558 auto constexpr Hcols = Hrows;
559 auto const heg = Hg.block<Hrows, Hcols>(0, g * Hcols);
560 for (auto j = 0; j < ElementType::kNodes; ++j)
561 for (auto dj = 0; dj < kDims; ++dj)
562 for (auto i = 0; i < ElementType::kNodes; ++i)
563 for (auto di = 0; di < kDims; ++di)
564 triplets.push_back(Triplet{
565 static_cast<SparseIndex>(kDims * nodes(i) + di),
566 static_cast<SparseIndex>(kDims * nodes(j) + dj),
567 heg(kDims * i + di, kDims * j + dj)});
568 }
569
570 auto const n = InputDimensions();
571 CSCMatrix H(n, n);
572 H.setFromTriplets(triplets.begin(), triplets.end());
573 return H;
574 }
575}

580 PBAT_PROFILE_NAMED_SCOPE("pbat.fem.HyperElasticPotential.ToVector");

619inline void HyperElasticPotential<TMesh, THyperElasticEnergy>::CheckValidState() const
620{
621 auto const numberOfQuadraturePoints = wg.size();
622 auto constexpr kExpectedGNegRows = ElementType::kNodes;
623 auto const expectedGNegCols = MeshType::kDims * numberOfQuadraturePoints;
624 bool const bShapeFunctionGradientsHaveCorrectDimensions =
625 (GNeg.rows() == kExpectedGNegRows) and (GNeg.cols() == expectedGNegCols);
626 if (not bShapeFunctionGradientsHaveCorrectDimensions)
627 {
628 std::string const what = fmt::format(
629 "Expected shape function gradients at element quadrature points of dimensions "
630 "|#nodes-per-element|={} x |#mesh-dims * #quad.pts.|={} for but got {}x{} instead",
631 kExpectedGNegRows,
632 expectedGNegCols,
633 GNeg.rows(),
634 GNeg.cols());
635 throw std::invalid_argument(what);
636 }
637 bool const bLameCoefficientsHaveCorrectDimensions =
638 (mug.size() == numberOfQuadraturePoints) and (lambdag.size() == numberOfQuadraturePoints);
639 if (not bLameCoefficientsHaveCorrectDimensions)
640 {
641 std::string const what = fmt::format(
642 "Expected quadrature point lame coefficients with dimensions {0}x1 and "
643 "{0}x1 for mug and lambdag, but got {1}x1 and {2}x1 instead.",
644 numberOfQuadraturePoints,
645 mug.size(),
646 lambdag.size());
647 throw std::invalid_argument(what);
