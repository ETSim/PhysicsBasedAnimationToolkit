197template <int Dims>
198template <class FVisit, class FStop>
199inline void KdTree<Dims>::BreadthFirstSearch(FVisit visit, FStop stop, Index root) const
200{
201 PBAT_PROFILE_NAMED_SCOPE("pbat.geometry.KdTree.BreadthFirstSearch");
202 std::queue<Index> bfs{};
203 bfs.push(root);
204 while (!bfs.empty())
205 {
206 auto const nodeIdx = bfs.front();
207 auto const nodeIdxStl = static_cast<std::size_t>(nodeIdx);
208 KdTreeNode const& node = mNodes[nodeIdxStl];
209 bfs.pop();
210 if (stop(nodeIdx, node))
211 break;
212
213 if (!visit(nodeIdx, node))
214 continue;
215
216 if (node.HasLeftChild())
217 bfs.push(node.lc);
218
219 if (node.HasRightChild())
220 bfs.push(node.rc);
221 }
222}

253template <class TDerivedP>
254inline void
255KdTree<Dims>::Construct(Eigen::DenseBase<TDerivedP> const& P, std::size_t maxPointsInLeaf)
256{
257 PBAT_PROFILE_NAMED_SCOPE("pbat.geometry.KdTree.Construct");
258 std::size_t const n = static_cast<std::size_t>(P.cols());
259 mNodes.clear();
260 mNodes.reserve(n);
261 auto iota = std::views::iota(Index{0}, static_cast<Index>(n));
262 mPermutation.assign(iota.begin(), iota.end());
263
264 geometry::AxisAlignedBoundingBox<Dims> const aabb{P};
265 auto const depth = 0;
266 auto const begin = 0;
267 auto const rootIdx = AddNode(begin, n, depth);
268 DoConstruct(rootIdx, P, aabb, begin, n, maxPointsInLeaf);
269}

271template <int Dims>
272template <class TDerivedP>
273inline void KdTree<Dims>::DoConstruct(
274 Index const nodeIdx,
275 Eigen::DenseBase<TDerivedP> const& P,
276 AxisAlignedBoundingBox<Dims> const& aabb,
277 Index const begin,
278 std::size_t const n,
279 std::size_t maxPointsInLeaf)
280{
281 if (n <= maxPointsInLeaf)
282 return;
283
284 Eigen::Index dimension{};
285 (aabb.max() - aabb.min()).maxCoeff(&dimension);
286
287 std::size_t const halfnStl = n / 2;
288 Index const halfn = static_cast<Index>(halfnStl);
289 std::size_t const nodeIdxStl = static_cast<std::size_t>(nodeIdx);
290
291 std::nth_element(
292 mPermutation.begin() + begin,
293 mPermutation.begin() + begin + halfn,
294 mPermutation.begin() + begin + static_cast<Index>(n),
295 [&](Index const lhs, Index const rhs) { return P(dimension, lhs) < P(dimension, rhs); });
296
297 auto const depth = mNodes[nodeIdxStl].depth;
298 auto lnode = AddNode(begin, halfnStl, depth + 1u);
299 auto rnode = AddNode(begin + halfn, n - halfnStl, depth + 1u);
300 mNodes[nodeIdxStl].lc = lnode;
301 mNodes[nodeIdxStl].rc = rnode;
302
303 Scalar const split = P(dimension, mPermutation[begin + halfnStl]);
304 AxisAlignedBoundingBox laabb = aabb;
305 laabb.max()(dimension) = split;
306 AxisAlignedBoundingBox raabb = aabb;
307 raabb.min()(dimension) = split;
308
309 DoConstruct(mNodes[nodeIdxStl].lc, P, laabb, begin, halfnStl, maxPointsInLeaf);
310 DoConstruct(mNodes[nodeIdxStl].rc, P, raabb, begin + halfn, n - halfnStl, maxPointsInLeaf);
311}
