115template <CElement TElement>
116class NodalKey
117{
118 public:
119 using SelfType = NodalKey<TElement>;
120 static int constexpr kVertices = TElement::AffineBaseType::kNodes;
121
122 NodalKey(
123 IndexVector<kVertices> const& cellVertices,
124 IndexVector<kVertices> const& sortOrder,
125 Eigen::Vector<math::Rational, kVertices> const& N)
126 : mCellVertices(cellVertices), mSortOrder(sortOrder), mN(N), mSize()
127 {
128 // Remove vertices whose corresponding shape function is zero
129 auto it = std::remove_if(mSortOrder.begin(), mSortOrder.end(), [this](Index o) {
130 return mN[o] == 0;
131 });
132 // Count number of non-zero shape functions into Size
133 mSize = static_cast<decltype(mSize)>(std::distance(mSortOrder.begin(), it));
134 }
135
136 bool operator==(SelfType const& rhs) const
137 {
138 // Sizes must match
139 if (mSize != rhs.mSize)
140 return false;
141 for (auto i = 0u; i < mSize; ++i)
142 {
143 // Vertices involved in affine map must match
144 Index const lhsVertex = mCellVertices[mSortOrder[i]];
145 Index const rhsVertex = rhs.mCellVertices[rhs.mSortOrder[i]];
146 if (lhsVertex != rhsVertex)
147 return false;
148 // Affine weights at matching vertices must match
149 math::Rational const lhsN = mN[mSortOrder[i]];
150 math::Rational const rhsN = rhs.mN[rhs.mSortOrder[i]];
151 if (lhsN != rhsN)
152 return false;
153 }
154 // Everything matches, (*this) and rhs must represent same node
155 return true;
156 }
157
158 bool operator<(SelfType const& rhs) const
159 {
160 // Sort by size first
161 if (mSize != rhs.mSize)
162 return mSize < rhs.mSize;
163 // Then sort by vertex indices
164 for (auto i = 0; i < mSize; ++i)
165 {
166 Index const lhsVertex = mCellVertices[mSortOrder[i]];
167 Index const rhsVertex = rhs.mCellVertices[rhs.mSortOrder[i]];
168 if (lhsVertex != rhsVertex)
169 return lhsVertex < rhsVertex;
170 }
171 // Then sort by coordinates
172 for (auto i = 0; i < mSize; ++i)
173 {
174 math::Rational const lhsN = mN[mSortOrder[i]];
175 math::Rational const rhsN = rhs.mN[rhs.mSortOrder[i]];
176 if (lhsN != rhsN)
177 return lhsN < rhsN;
178 }
179 // (*this) == rhs is true, so (*this) is not less than rhs
180 return false;
181 }
182
183 private:
184 IndexVector<kVertices> mCellVertices;
185 IndexVector<kVertices> mSortOrder;
186 Eigen::Vector<math::Rational, kVertices> mN;
187 int mSize;
188};

192template <CElement TElement, int Dims>
193Mesh<TElement, Dims>::Mesh(
194 Eigen::Ref<MatrixX const> const& V,
195 Eigen::Ref<IndexMatrixX const> const& C)
196{
197 PBAT_PROFILE_NAMED_SCOPE("pbat.fem.Mesh.Construct");
198
199 // Smart nodal indexing is only relevant for higher-order meshes
200 if constexpr (kOrder == 1)
201 {
202 X = V;
203 E = C;
204 return;
205 }
206 else
207 {
208 using AffineElementType = typename ElementType::AffineBaseType;
209 auto constexpr kVerticesPerCell = AffineElementType::kNodes;
210
211 static_assert(
212 kDims >= ElementType::kDims,
213 "Element TElement does not exist in Dims dimensions");
214 assert(C.rows() == kVerticesPerCell);
215 assert(V.rows() == kDims);
216
217 using detail::NodalKey;
218 using NodeMap = std::map<NodalKey<ElementType>, Index>;
219
220 auto const numberOfCells = C.cols();
221 auto const numberOfVertices = V.cols();
222
223 NodeMap nodeMap{};
224 std::vector<Vector<kDims>> nodes{};
225 nodes.reserve(static_cast<std::size_t>(numberOfVertices));
226
227 // Construct mesh topology, i.e. assign mesh nodes to elements,
228 // ensuring that adjacent elements share their common nodes.
229 E.resize(ElementType::kNodes, numberOfCells);
230 for (auto c = 0; c < numberOfCells; ++c)
231 {
232 IndexVector<kVerticesPerCell> const cellVertices = C.col(c);
233 Matrix<kDims, kVerticesPerCell> const Xc = V(Eigen::placeholders::all, cellVertices);
234
235 // Sort based on cell vertex index
236 IndexVector<kVerticesPerCell> sortOrder{};
237 std::iota(sortOrder.begin(), sortOrder.end(), 0);
238 std::ranges::sort(sortOrder, [&](Index i, Index j) {
239 return cellVertices[i] < cellVertices[j];
240 });
241 // Loop over nodes of element and create the node on first visit
242 auto const nodalCoordinates = common::ToEigen(ElementType::Coordinates)
243 .reshaped(ElementType::kDims, ElementType::kNodes)
244 .template cast<math::Rational>() /
245 ElementType::kOrder;
246 for (auto i = 0; i < nodalCoordinates.cols(); ++i)
247 {
248 // Use exact rational arithmetic to evaluate affine element shape functions at the
249 // node to get its exact affine coordinates
250 auto const Xi = nodalCoordinates.col(i);
251 auto const N = AffineElementType::N(Xi);
252 NodalKey<ElementType> const key{cellVertices, sortOrder, N};
253 auto it = nodeMap.find(key);
254 bool const bNodeAlreadyCreated = it != nodeMap.end();
255 if (!bNodeAlreadyCreated)
256 {
257 auto const nodeIdx = static_cast<Index>(nodes.size());
258 Vector<kDims> const xi = Xc * N.template cast<Scalar>();
259 nodes.push_back(xi);
260 bool bInserted{};
261 std::tie(it, bInserted) = nodeMap.insert({key, nodeIdx});
262 assert(bInserted);
263 }
264 Index const node = it->second;
265 E(i, c) = node;
266 }
267 }
268 // Collect node positions
269 X = common::ToEigen(nodes);
270 }
271}

275template <CElement TElement, int Dims, int QuadratureOrder, class TDerivedX, class TDerivedE>
276inline MatrixX
277MeshQuadraturePoints(Eigen::MatrixBase<TDerivedX> const& X, Eigen::MatrixBase<TDerivedE> const& E)
278{
279 using ElementType = TElement;
280 using AffineElementType = typename ElementType::AffineBaseType;
281 using QuadratureRuleType = typename ElementType::template QuadratureType<QuadratureOrder>;
282 auto constexpr kQuadPts = QuadratureRuleType::kPoints;
283 auto const numberOfElements = E.cols();
284 auto const XgRef = common::ToEigen(QuadratureRuleType::points)
285 .reshaped(QuadratureRuleType::kDims + 1, kQuadPts);
286 MatrixX Xg(Dims, numberOfElements * kQuadPts);
287 for (auto e = 0; e < numberOfElements; ++e)
288 {
289 auto const nodes = E.col(e);
290 auto const vertices = nodes(ElementType::Vertices);
291 auto constexpr kRowsJ = Dims;
292 auto constexpr kColsJ = AffineElementType::kNodes;
293 Matrix<kRowsJ, kColsJ> const Ve = X(Eigen::placeholders::all, vertices);
294 for (auto g = 0; g < kQuadPts; ++g)
295 {
296 Xg.col(e * kQuadPts + g) = Ve * XgRef.col(g);
297 }
298 }
299 return Xg;
300}

331 Eigen::Ref<MatrixX const> const& V,
332 Eigen::Ref<IndexMatrixX const> const& C)
333 : X(V), E(C)
334{
335 static_assert(TElement::kOrder == 1, "TElement must be a linear element");
336 auto const nNodesPerElement = C.rows();
337 if (nNodesPerElement != TElement::kNodes)
338 {
339 throw std::invalid_argument(fmt::format(
340 "Expected {}x{} elements, but got {}x{}",
341 TElement::kNodes,
342 C.cols(),
343 C.rows(),
344 C.cols()));
345 }
346 if (V.rows() != Dims)
347 {
348 throw std::invalid_argument(
349 fmt::format("Expected {}x{} nodes, but got {}x{}", Dims, V.cols(), V.rows(), V.cols()));
350 }
351 if (V.rows() < TElement::kDims)
352 {
353 throw std::invalid_argument(fmt::format(
354 "Nodal coordinates must have dimensions > {} for the requested element",
355 TElement::kDims));
356 }
357}
