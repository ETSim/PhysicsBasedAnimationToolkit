65template <auto kBlockThreads>
66__global__ void MinimizeBackwardEuler(BackwardEulerMinimization BDF)
67{
68 // Get thread info
69 using BlockReduce = typename BackwardEulerMinimization::BlockReduce<kBlockThreads>;
70 using BlockStorage = typename BackwardEulerMinimization::BlockStorage<kBlockThreads>;
71 extern __shared__ __align__(alignof(BlockStorage)) char shared[];
72 auto tid = threadIdx.x;
73 auto bid = blockIdx.x;
74
75 // Get vertex-tet adjacency information
76 GpuIndex i = BDF.partition[bid]; // Vertex index
77 GpuIndex GVTbegin = BDF.GVTp[i];
78 GpuIndex nAdjacentElements = BDF.GVTp[i + 1] - GVTbegin;
79 // 1. Compute vertex-element elastic energy derivatives w.r.t. i and store them in shared memory
80 SMatrix<GpuScalar, 3, 4> Hgt = Zeros<GpuScalar, 3, 4>();
81 auto Ht = Hgt.Slice<3, 3>(0, 0);
82 auto gt = Hgt.Col(3);
83 for (auto elocal = tid; elocal < nAdjacentElements; elocal += kBlockThreads)
84 {
85 GpuIndex e = BDF.GVTn[GVTbegin + elocal];
86 GpuIndex ilocal = BDF.GVTilocal[GVTbegin + elocal];
87 SVector<GpuIndex, 4> Te = FromBuffers<4, 1>(BDF.T, e);
88 SMatrix<GpuScalar, 4, 3> GPe = FromFlatBuffer<4, 3>(BDF.GP, e);
89 SMatrix<GpuScalar, 3, 4> xe = FromBuffers(BDF.x, Te.Transpose());
90 SVector<GpuScalar, 2> lamee = FromFlatBuffer<2, 1>(BDF.lame, e);
91 GpuScalar wg = BDF.wg[e];
92 SMatrix<GpuScalar, 3, 3> Fe = xe * GPe;
93 pbat::physics::StableNeoHookeanEnergy<3> Psi{};
94 SVector<GpuScalar, 9> gF;
95 SMatrix<GpuScalar, 9, 9> HF;
96 Psi.gradAndHessian(Fe, lamee(0), lamee(1), gF, HF);
97 using pbat::sim::vbd::kernels::AccumulateElasticGradient;
98 using pbat::sim::vbd::kernels::AccumulateElasticHessian;
99 AccumulateElasticHessian(ilocal, wg, GPe, HF, Ht);
100 AccumulateElasticGradient(ilocal, wg, GPe, gF, gt);
101 }
102
103 // 2. Compute total vertex hessian and gradient via parallel reduction
104 SMatrix<GpuScalar, 3, 4> Hgi = BlockReduce(reinterpret_cast<BlockStorage&>(shared)).Sum(Hgt);
105 if (tid > 0)
106 return;
107 auto Hi = Hgi.Slice<3, 3>(0, 0);
108 auto gi = Hgi.Col(3);
109
110 // 3. Add stiffness damping
111 GpuScalar mi = BDF.m[i];
112 SVector<GpuScalar, 3> xti = FromBuffers<3, 1>(BDF.xt, i);
113 SVector<GpuScalar, 3> xi = FromBuffers<3, 1>(BDF.x, i);
114 using pbat::sim::vbd::kernels::AddDamping;
115 AddDamping(BDF.dt, xti, xi, BDF.kD, gi, Hi);
116
117 // 3. Add contact energy
118 static auto constexpr kMaxContacts = BackwardEulerMinimization::kMaxCollidingTrianglesPerVertex;
119 SVector<GpuIndex, kMaxContacts> f = FromFlatBuffer<kMaxContacts, 1>(BDF.fc, i);
120 auto nContacts = Dot(Ones<GpuIndex, kMaxContacts>(), f >= 0);
121 SVector<GpuScalar, kMaxContacts> fa = Zeros<GpuIndex, kMaxContacts>();
122 for (auto c = 0; c < nContacts; ++c)
123 fa(c) = BDF.FA[f(c)];
124 auto sumfa = Dot(fa, Ones<GpuScalar, kMaxContacts>());
125 GpuScalar muC = (BDF.XVA[i] * BDF.muC) / sumfa;
126 for (auto c = 0; c < nContacts; ++c)
127 {
128 using pbat::sim::vbd::kernels::AccumulateVertexTriangleContact;
129 auto finds = FromBuffers<3, 1>(BDF.F, f(c));
130 auto xtf = FromBuffers(BDF.xt, finds.Transpose());
131 auto xf = FromBuffers(BDF.x, finds.Transpose());
132 AccumulateVertexTriangleContact(
133 xti,
134 xi,
135 xtf,
136 xf,
137 BDF.dt,
138 fa(c) * muC,
139 BDF.muF,
140 BDF.epsv,
141 gi,
142 Hi);
143 }
144
145 // 4. Add inertial term
146 SVector<GpuScalar, 3> xitilde = FromBuffers<3, 1>(BDF.xtilde, i);
147 using pbat::sim::vbd::kernels::AddInertiaDerivatives;
148 AddInertiaDerivatives(BDF.dt2, mi, xitilde, xi, gi, Hi);
149
150 // 5. Integrate positions
151 using pbat::sim::vbd::kernels::IntegratePositions;
152 IntegratePositions(gi, Hi, xi, BDF.detHZero);
153 ToBuffers(xi, BDF.xb, i);
154};
