110template <CMesh TMesh, int QuadratureOrder>
111inline VectorX LoadVector<TMesh, QuadratureOrder>::ToVector() const
112{
113 PBAT_PROFILE_NAMED_SCOPE("pbat.fem.LoadVector.ToVector");
114 CheckValidState();
115 auto const n = mesh.X.cols() * dims;
116 auto const numberOfElements = mesh.E.cols();
117 VectorX f = VectorX::Zero(n);
118 for (auto e = 0; e < numberOfElements; ++e)
119 {
120 auto const nodes = mesh.E.col(e);
121 for (auto i = 0; i < nodes.size(); ++i)
122 {
123 f.segment(dims * nodes(i), dims) += N(i, e) * fe.col(e);
124 }
125 }
126 return f;
127}

140template <CMesh TMesh, int QuadratureOrder>
141template <class TDerived>
142inline void LoadVector<TMesh, QuadratureOrder>::SetLoad(Eigen::DenseBase<TDerived> const& load)
143{
144 auto const numberOfElements = mesh.E.cols();
145 if (load.rows() != dims)
146 {
147 std::string const what = fmt::format(
148 "LoadVector<TMesh,{0}> discretizes a {0}-dimensional load, but received "
149 "{1}-dimensional input load",
150 dims,
151 load.rows());
152 throw std::invalid_argument(what);
153 }
154 if (load.cols() != 1 && load.cols() != numberOfElements)
155 {
156 std::string const what = fmt::format(
157 "Input load vector must be constant or piecewise element constant, but size was {}",
158 load.cols());
159 throw std::invalid_argument(what);
160 }
161
162 fe.resize(dims, numberOfElements);
163 if (load.cols() == 1)
164 fe.colwise() = load.col(0);
165 else // load.cols() == numberOfElements
166 fe = load;
167}
