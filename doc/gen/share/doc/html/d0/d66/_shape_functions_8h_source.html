38template <CElement TElement, int QuadratureOrder>
39auto ShapeFunctions()
40 -> Matrix<TElement::kNodes, TElement::template QuadratureType<QuadratureOrder>::kPoints>
41{
42 using QuadratureRuleType = typename TElement::template QuadratureType<QuadratureOrder>;
43 using ElementType = TElement;
44 auto const Xg = common::ToEigen(QuadratureRuleType::points)
45 .reshaped(QuadratureRuleType::kDims + 1, QuadratureRuleType::kPoints)
46 .template bottomRows<QuadratureRuleType::kDims>();
47 Matrix<ElementType::kNodes, QuadratureRuleType::kPoints> Ng{};
48 for (auto g = 0; g < QuadratureRuleType::kPoints; ++g)
49 {
50 Ng.col(g) = ElementType::N(Xg.col(g));
51 }
52 return Ng;
53}

63template <int QuadratureOrder, CMesh TMesh>
64CSRMatrix ShapeFunctionMatrix(TMesh const& mesh)
65{
66 PBAT_PROFILE_NAMED_SCOPE("pbat.fem.ShapeFunctionMatrix");
67 using ElementType = typename TMesh::ElementType;
68 auto const Ng = ShapeFunctions<ElementType, QuadratureOrder>();
69 auto const numberOfNodes = mesh.X.cols();
70 auto const numberOfElements = mesh.E.cols();
71 auto const numberOfQuadPoints = Ng.cols();
72 auto const m = numberOfQuadPoints * numberOfElements;
73 auto const n = numberOfNodes;
74 auto constexpr kNodesPerElement = ElementType::kNodes;
75 CSRMatrix N(m, n);
76 N.reserve(IndexVectorX::Constant(m, kNodesPerElement));
77 for (auto e = 0; e < numberOfElements; ++e)
78 {
79 auto const nodes = mesh.E.col(e);
80 for (auto g = 0; g < numberOfQuadPoints; ++g)
81 {
82 auto const row = e * numberOfQuadPoints + g;
83 for (auto i = 0; i < Ng.rows(); ++i)
84 {
85 auto const col = nodes(i);
86 N.insert(row, col) = Ng(i, g);
87 }
88 }
89 }
90 return N;
91}

205template <int QuadratureOrder, CMesh TMesh, class TDerived>
206MatrixX IntegratedShapeFunctions(TMesh const& mesh, Eigen::DenseBase<TDerived> const& detJe)
207{
208 PBAT_PROFILE_NAMED_SCOPE("pbat.fem.IntegratedShapeFunctions");
209 using MeshType = TMesh;
210 using ElementType = typename MeshType::ElementType;
211 using QuadratureRuleType = typename ElementType::template QuadratureType<QuadratureOrder>;
212 auto constexpr kQuadPts = QuadratureRuleType::kPoints;
213 auto constexpr kQuadratureOrder = QuadratureOrder;
214 auto const numberOfElements = mesh.E.cols();
215 bool const bHasDeterminants = (detJe.rows() == kQuadPts) and (detJe.cols() == numberOfElements);
216 if (not bHasDeterminants)
217 {
218 std::string const what = fmt::format(
219 "Expected element jacobian determinants of dimensions {}x{} for element quadrature of "
220 "order={}, but got {}x{}",
221 kQuadPts,
222 numberOfElements,
223 kQuadratureOrder,
224 detJe.rows(),
225 detJe.cols());
226 throw std::invalid_argument(what);
227 }
228 // Precompute element shape functions
229 auto constexpr kNodesPerElement = ElementType::kNodes;
230 Matrix<kNodesPerElement, kQuadPts> const Ng = ShapeFunctions<ElementType, kQuadratureOrder>();
231 // Integrate shape functions
232 MatrixX N = MatrixX::Zero(kNodesPerElement, numberOfElements);
233 auto const wg = common::ToEigen(QuadratureRuleType::weights);
234 tbb::parallel_for(Index{0}, Index{numberOfElements}, [&](Index e) {
235 for (auto g = 0; g < QuadratureRuleType::kPoints; ++g)
236 {
237 N.col(e) += (wg(g) * detJe(g, e)) * Ng.col(g);
238 }
239 });
240 return N;
241}

301template <CElement TElement, class TDerivedXi, class TDerivedX>
302auto ShapeFunctionGradients(
303 Eigen::MatrixBase<TDerivedXi> const& Xi,
304 Eigen::MatrixBase<TDerivedX> const& X) -> Matrix<TElement::kNodes, TDerivedX::RowsAtCompileTime>
305{
306 auto constexpr kInputDims = TElement::kDims;
307 auto constexpr kOutputDims = TDerivedX::RowsAtCompileTime;
308 auto constexpr kNodes = TElement::kNodes;
309 using AffineElementType = typename TElement::AffineBaseType;
310 Matrix<kNodes, kInputDims> const GN = TElement::GradN(Xi);
311 Matrix<kOutputDims, kInputDims> J;
312 Matrix<kNodes, kOutputDims> GP;
313
314 bool constexpr bIsElementLinear = std::is_same_v<TElement, AffineElementType>;
315 if constexpr (bIsElementLinear)
316 {
317 J = X * GN;
318 }
319 else
320 {
321 J = X * AffineElementType::GradN(Xi);
322 }
323 bool constexpr bIsJacobianSquare = kInputDims == kOutputDims;
324 if (bIsJacobianSquare)
325 {
326 GP.transpose() = J.transpose().fullPivLu().solve(GN.transpose());
327 }
328 else
329 {
330 Matrix<kInputDims, kInputDims> const JTJ = J.transpose() * J;
331 GP.transpose() = J * JTJ.ldlt().solve(GN.transpose());
332 }
333 return GP;
334}

344template <int QuadratureOrder, CMesh TMesh>
345MatrixX ShapeFunctionGradients(TMesh const& mesh)
346{
347 PBAT_PROFILE_NAMED_SCOPE("pbat.fem.ShapeFunctionGradients");
348 using MeshType = TMesh;
349 using ElementType = typename MeshType::ElementType;
350 using QuadratureRuleType = typename ElementType::template QuadratureType<QuadratureOrder>;
351 using AffineElementType = typename ElementType::AffineBaseType;
352 auto const numberOfElements = mesh.E.cols();
353 auto constexpr kNodesPerElement = ElementType::kNodes;
354 auto const Xg = common::ToEigen(QuadratureRuleType::points)
355 .reshaped(QuadratureRuleType::kDims + 1, QuadratureRuleType::kPoints)
356 .template bottomRows<ElementType::kDims>();
357 MatrixX GNe(kNodesPerElement, numberOfElements * MeshType::kDims * QuadratureRuleType::kPoints);
358 tbb::parallel_for(Index{0}, Index{numberOfElements}, [&](Index e) {
359 auto const nodes = mesh.E.col(e);
360 auto const vertices = nodes(ElementType::Vertices);
361 auto constexpr kRowsJ = MeshType::kDims;
362 auto constexpr kColsJ = AffineElementType::kNodes;
363 Matrix<kRowsJ, kColsJ> const Ve = mesh.X(Eigen::placeholders::all, vertices);
364 for (auto g = 0; g < QuadratureRuleType::kPoints; ++g)
365 {
366 auto const GP = ShapeFunctionGradients<ElementType>(Xg.col(g), Ve);
367 auto constexpr kStride = MeshType::kDims * QuadratureRuleType::kPoints;
368 GNe.block<kNodesPerElement, MeshType::kDims>(0, e * kStride + g * MeshType::kDims) = GP;
369 }
370 });
371 return GNe;
372}
