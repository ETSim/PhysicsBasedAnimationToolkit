134template <CMesh TMesh>
135inline CSCMatrix Gradient<TMesh>::ToMatrix() const
136{
137 PBAT_PROFILE_NAMED_SCOPE("pbat.fem.Gradient.ToMatrix");
138 using SparseIndex = typename CSCMatrix::StorageIndex;
139 using Triplet = Eigen::Triplet<Scalar, SparseIndex>;
140
141 std::vector<Triplet> triplets{};
142 triplets.reserve(static_cast<std::size_t>(GNeg.size()));
143 auto const numberOfQuadraturePoints = eg.size();
144 auto constexpr kNodesPerElement = ElementType::kNodes;
145 for (auto g = 0; g < numberOfQuadraturePoints; ++g)
146 {
147 auto const e = eg(g);
148 auto const nodes = mesh.E.col(e);
149 auto const Geg = GNeg.block<kNodesPerElement, kDims>(0, g * kDims);
150 for (auto d = 0; d < kDims; ++d)
151 {
152 for (auto j = 0; j < kNodesPerElement; ++j)
153 {
154 auto const ni = static_cast<SparseIndex>(d * numberOfQuadraturePoints + g);
155 auto const nj = static_cast<SparseIndex>(nodes(j));
156 triplets.push_back(Triplet(ni, nj, Geg(j, d)));
157 }
158 }
159 }
160 CSCMatrix G(OutputDimensions(), InputDimensions());
161 G.setFromTriplets(triplets.begin(), triplets.end());
162 return G;
163}

165template <CMesh TMesh>
166inline void Gradient<TMesh>::CheckValidState() const
167{
168 auto const numberOfQuadraturePoints = eg.size();
169 auto constexpr kExpectedGNegRows = ElementType::kNodes;
170 auto const expectedGNegCols = kDims * numberOfQuadraturePoints;
171 bool const bShapeFunctionGradientsHaveCorrectDimensions =
172 (GNeg.rows() == kExpectedGNegRows) and (GNeg.cols() == expectedGNegCols);
173 if (not bShapeFunctionGradientsHaveCorrectDimensions)
174 {
175 std::string const what = fmt::format(
176 "Expected shape function gradients at element quadrature points of dimensions "
177 "|#nodes-per-element|={} x |#mesh-dims * #quad.pts.|={} for polynomial but got {}x{} "
178 "instead",
179 kExpectedGNegRows,
180 expectedGNegCols,
181 GNeg.rows(),
182 GNeg.cols());
183 throw std::invalid_argument(what);
184 }
185}

187template <CMesh TMesh>
188template <class TDerivedIn, class TDerivedOut>
189inline void Gradient<TMesh>::Apply(
190 Eigen::MatrixBase<TDerivedIn> const& x,
191 Eigen::DenseBase<TDerivedOut>& y) const
192{
193 PBAT_PROFILE_NAMED_SCOPE("pbat.fem.Gradient.Apply");
194 // Check inputs
195 bool const bDimensionsMatch = (x.cols() == y.cols()) and (x.rows() == InputDimensions()) and
196 (y.rows() == OutputDimensions());
197 if (not bDimensionsMatch)
198 {
199 std::string const what = fmt::format(
200 "Expected input to have rows={} and output to have rows={}, and same number of "
201 "columns, but got dimensions "
202 "x,y=({},{}), ({},{})",
203 InputDimensions(),
204 OutputDimensions(),
205 x.rows(),
206 x.cols(),
207 y.rows(),
208 y.cols());
209 throw std::invalid_argument(what);
210 }
211 // Compute gradient
212 auto constexpr kNodesPerElement = ElementType::kNodes;
213 auto const numberOfQuadraturePoints = eg.size();
214 for (auto c = 0; c < x.cols(); ++c)
215 {
216 for (auto g = 0; g < numberOfQuadraturePoints; ++g)
217 {
218 auto const e = eg(g);
219 auto const nodes = mesh.E.col(e);
220 auto const xe = x.col(c)(nodes);
221 auto const Geg = GNeg.block<kNodesPerElement, kDims>(0, g * kDims);
222 for (auto d = 0; d < kDims; ++d)
223 {
224 y(d * numberOfQuadraturePoints + g) +=
225 Geg(Eigen::placeholders::all, d).transpose() * xe;
226 }
227 }
228 }
229}
