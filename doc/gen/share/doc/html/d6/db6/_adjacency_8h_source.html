57template <
58 class TWeightedEdgeIterator,
59 class TWeightedEdge = typename std::iterator_traits<TWeightedEdgeIterator>::value_type,
60 class TScalar = typename WeightedEdgeTraits<TWeightedEdge>::ScalarType,
61 class TIndex = typename WeightedEdgeTraits<TWeightedEdge>::IndexType>
62auto AdjacencyMatrixFromEdges(
63 TWeightedEdgeIterator begin,
64 TWeightedEdgeIterator end,
65 TIndex m = TIndex(-1),
66 TIndex n = TIndex(-1)) -> Eigen::SparseMatrix<TScalar, Eigen::RowMajor, TIndex>
67{
68 if (m < 0)
69 {
70 m = std::max_element(
71 begin,
72 end,
73 [](TWeightedEdge const& lhs, TWeightedEdge const& rhs) {
74 return lhs.row() < rhs.row();
75 })
76 ->row() +
77 TIndex(1);
78 }
79 if (n < 0)
80 {
81 n = std::max_element(
82 begin,
83 end,
84 [](TWeightedEdge const& lhs, TWeightedEdge const& rhs) {
85 return lhs.col() < rhs.col();
86 })
87 ->col() +
88 TIndex(1);
89 }
90 Eigen::SparseMatrix<TScalar, Eigen::RowMajor, TIndex> G(m, n);
91 G.setFromTriplets(begin, end);
92 return G;
93}

163template <class TDerivedP, std::integral TIndex = typename TDerivedP::Scalar>
164auto MapToAdjacency(Eigen::DenseBase<TDerivedP> const& p, TIndex n = TIndex(-1))
165 -> std::tuple<Eigen::Vector<TIndex, Eigen::Dynamic>, Eigen::Vector<TIndex, Eigen::Dynamic>>
166{
167 using IndexVectorType = Eigen::Vector<TIndex, Eigen::Dynamic>;
168 if (n < 0)
169 n = p.maxCoeff() + TIndex(1);
170 auto s = common::Counts(p.begin(), p.end(), n);
171 auto ptr = common::CumSum(s);
172 auto adj = common::ArgSort(p.size(), [&](auto i, auto j) { return p(i) < p(j); });
173 return std::make_tuple(ptr, adj);
174}

225template <class TIndex = Index>
226auto ListOfListsToAdjacency(std::vector<std::vector<TIndex>> const& lil)
227{
228 auto n = static_cast<TIndex>(lil.size());
229 TIndex nEdges{0};
230 using IndexVectorType = Eigen::Vector<TIndex, Eigen::Dynamic>;
231 IndexVectorType ptr(n + TIndex(1));
232 ptr(0) = TIndex(0);
233 for (auto l = 0; l < n; ++l)
234 {
235 auto lStl = static_cast<std::size_t>(l);
236 auto nVerticesInPartition = static_cast<TIndex>(lil[lStl].size());
237 ptr(l + 1) = ptr(l) + nVerticesInPartition;
238 nEdges += nVerticesInPartition;
239 }
240 IndexVectorType adj(nEdges);
241 for (auto l = 0; l < n; ++l)
242 {
243 auto start = ptr(l);
244 auto end = ptr(l + 1);
245 auto lStl = static_cast<std::size_t>(l);
246 adj(Eigen::seqN(start, end - start)) = common::ToEigen(lil[lStl]);
247 }
248 return std::make_tuple(ptr, adj);
249}

295template <
296 class TDerivedPtr,
297 class TDerivedAdj,
298 class TIndex = typename TDerivedPtr::Scalar,
299 class Func>
300void RemoveEdges(
301 Eigen::DenseBase<TDerivedPtr>& ptr,
302 Eigen::DenseBase<TDerivedAdj>& adj,
303 Func fShouldDeleteEdge)
304{
305 auto nPartitions = ptr.size() - 1;
306 using IndexVectorType = Eigen::Vector<TIndex, Eigen::Dynamic>;
307 IndexVectorType sizes(nPartitions);
308 sizes.setZero();
309 TIndex nEdges(0);
310 // Compress edges by discarding edges to remove
311 IndexVectorType adjTmp(adj.size());
312 IndexVectorType ptrTmp = ptr;
313 for (auto p = 0, kk = 0; p < nPartitions; ++p)
314 {
315 auto kBegin = ptr(p);
316 auto kEnd = ptr(p + TIndex(1));
317 sizes(p) = kEnd - kBegin;
318 for (auto k = kBegin; k < kEnd; ++k)
319 {
320 if (fShouldDeleteEdge(p, adj(k)))
321 {
322 --sizes(p);
323 }
324 else
325 {
326 adjTmp(kk) = adj(k);
327 ++kk;
328 }
329 }
330 nEdges += sizes(p);
331 ptrTmp(p + TIndex(1)) = ptrTmp(p) + sizes(p);
332 }
333 ptr = ptrTmp;
334 adj = adjTmp.head(nEdges);
335}
